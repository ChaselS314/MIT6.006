##### Insertion Sort

  > for j <- 2 to n
  > ```c++
  > do key <- A[j]
  > 	i <- j-1
  > while i > 0 and A[i] > key:
  > 	do A[i+1] <- A[i]
  > 		i = i - 1
  > A[i+1] <- key
  > ```

> 		done.

>
>
> ```c++
> Merge(l1, l2, l)
>     pl1 <- 1, pl2 <- 1, pl <- 1
>     while pl1 < len(l1) and pl2 < len(l2)
>         if l1[pl1] < l2[pl2]
>             l[pl] <- l1[pl1]
>             pl <- pl + 1, pl1 <- pl1 + 1
>         else
>             l[pl] <= l2[pl2]
>             pl <- pl + 1, pl2 <- pl2 + 1
>     while pl1 < len(l1)
>         l[pl] <- l1[pl1]
>         pl <- pl + 1, pl1 <- pl1 + 1
>     while pl2 < len(l2)
>         l[pl] <- l2[pl2]
>         pl <- pl + 1, pl2 <- pl2 + 1
> ```
